######### TESTING CLASSIFIERS ###########

# Initialize classifiers
classifiers = {
    'Logistic Regression': LogisticRegression(max_iter=200),
    'Random Forest': RandomForestClassifier(),
    'SVM': SVC(),
    'KNN': KNeighborsClassifier()
}

# Train and evaluate each classifier
results = {}
for name, clf in classifiers.items():
    clf.fit(X_train, y_train)
    y_pred = clf.predict(X_test)
    accuracy = accuracy_score(y_test, y_pred)
    results[name] = accuracy

# Sort classifiers by performance
sorted_results = sorted(results.items(), key=lambda x: x[1], reverse=True)
print("\nClassifiers sorted by performance:\n")
for name, accuracy in sorted_results:
    print(f"{name}: {accuracy}")

########### PARAMETER TUNING ############

# Define parameter grids for hyperparameter tuning
param_grid_rf = {
    'n_estimators': [10, 50, 100],
    'max_depth': [None, 10, 20, 30]
}

param_grid_svm = {
    'C': [0.1, 1, 10],
    'kernel': ['linear', 'rbf']
}

# Initialize GridSearchCV
grid_rf = GridSearchCV(estimator=RandomForestClassifier(), param_grid=param_grid_rf, cv=5)
grid_svm = GridSearchCV(estimator=SVC(), param_grid=param_grid_svm, cv=5)

# Fit GridSearchCV
grid_rf.fit(X_train, y_train)
grid_svm.fit(X_train, y_train)

# Display best parameters and scores
print("\n")
print("Best parameters for Random Forest:", grid_rf.best_params_)
print("Best score for Random Forest:", grid_rf.best_score_)

print("Best parameters for SVM:", grid_svm.best_params_)
print("Best score for SVM:", grid_svm.best_score_)

##### VISUALIZING MODEL PERFORMANCE #####

# Evaluate best model (e.g., Random Forest)
best_model = grid_rf.best_estimator_
y_pred = best_model.predict(X_test)

# Confusion matrix
conf_matrix = confusion_matrix(y_test, y_pred)

# Create ConfusionMatrixDisplay object
labels = ['Setosa','Versicolor','Virginica']
disp = ConfusionMatrixDisplay(conf_matrix, display_labels=labels)

# Plot the confusion matrix
plt.figure(figsize=(8, 6))
disp.plot(cmap='Blues', values_format='d', include_values=True)
plt.title('Confusion Matrix')
plt.show()

# Classification report
print("Classification Report:")
print(classification_report(y_test, y_pred, target_names=labels))
